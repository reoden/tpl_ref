\subsection{ex-Euler Theorem}
\begin{small}
$$
a^b=
\left\{  
             \begin{array}{lr}  
             a^{b\ mod\ \varphi(p)},              & gcd(a,p) = 1\\ 
             a^b,                                 & gcd(a,p) \neq 1, b < \varphi(p)\\
             a^{(b\ mod\ \varphi(p))+\varphi(p)}, & gcd(a,p) \neq 1, b \ge \varphi(p)
             \end{array}  
\right.
$$
\end{small}
